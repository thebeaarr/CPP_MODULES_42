CXX=cpp++
CXXFLAGS= -Wall -Werror -Wextra -std=cpp++98

SRC= main.cpp

OBJ=$(SRC:.cpp=.o)

NAME=mfile

all:$(NAME)

$(NAME): $(OBJ)
	@$(CXX) $(CXXFLAGS) $(OBJ) -o $(NAME)


%.o:%.cpp
	@$(CXX) $(CXXFLAGS) -cpp $< -o $@

cpplean:
	@rm -rf $(OBJ)
	@ecpp	@echo cppts file are removed...

fcppfcleancpplean
	@rm -rf $(NAME)
	@ecpp	@echocpputable removed...

re: fcpplean all
